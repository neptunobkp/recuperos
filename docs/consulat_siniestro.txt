SELECT SELE.NSINIE_ELE NUMEROSINIESTRO,
       SELE.FINSND_ELE DIADENUNCIO,
       SELE.FINSNM_ELE MESDENUNCIO,
       SELE.FINSNA_ELE ANIODENUNCIO,
       SIN.FSINID_SIN  DIASINIESTRO,
       SIN.FSINIM_SIN  MESSINIESTRO,
       SIN.FSINIA_SIN  ANIOSINIESTRO,
       SELE.CCAUES_ELE CAUSAL,
       SELE.CSUCUR_ELE CODIGOSUCURSAL,
       SELE.CTPDOC_ELE TIPODOCUMENTO,
       SELE.NPOLIZ_ELE NUMEROPOLIZA,
       CTPVEH_T05      CODIGOTIPOVEHICULO,
       CMARCA_T05      MARCA,
       CMODEL_T05      MODELO,
       CVERSION_T05    VERSION,
       FANOFA_T05      FECHAANIOFABRICACION,
       NMOTOR_T05      NUMEROMOTOR,
       NPATEN_T05      PATENTE,
       XCOLOR_T05      COLOR,
       XCHASS_T05      NUMEROCHASIS,
       MAE.XPERSO_MAE  NOMBREASEGURADO,
       MAE.XDICOM_MAE  DIRECCIONASEGURADO,
       MAE.CCOMUN_MAE  CODIGOCOMUNAASEGURADO,
       MAE.CCIUDA_MAE  CODIGOCIUDADASEGURADO,
       MAE.NFOCOM_MAE  TELEFONO1ASEGURADO,
       MAE.NFOPST_MAE  TELEFONO2ASEGURADO,
       RCP.XNODEN_RCP  NOMBREDENUNCIANTE,
       RCP.XDIDEN_RCP  DIRECCIONDENUNCIANTE,
       RCP.CCIUDA_RCP  CODIGOCIUDADENUNCIANTE,
       RCP.NRUDEN_RCP  RUTDENUNCIANTE,
       RCP.NTEDEN1_RCP TELEFONO1DENUNCIANTE,
       RCP.NTEDEN2_RCP TELEFONO2DENUNCIANTE,
       CDT.NRUCON_CDT  AS RUTCONDUCTOR,
       CDT.XCONDU_CDT  AS NOMBRECONDUCTOR,
       CDT.XPRFCO_CDT  AS NUMEROTELEFONOCONDUCTOR,
       CDT.XDOCON_CDT  AS DIRECCIONCONDUCTOR,
       CDT.CEMAIL_CDT  AS EMAILCONDUCTOR,
       SACC.XCASIN_ACC AS CALLEACCIDENTE,
       SACC.CCOMUN_ACC AS CODICOCOMUNAACCIDETE,
       SACC.CCIUDA_ACC AS CODIGOCIUDADACCIDENTE,
       SACC.IALCOH_ACC AS ACOHOLEMIAACCIDENTE
  FROM CONCORDE.SNACSELE SELE
       JOIN CONCORDE.SNACSSIN SIN 
       ON SELE.NSINIE_ELE = SIN.NSINIE_SIN
       JOIN CONCORDE.SUACST05 T05
          ON     T05.NDOCTO_T05 = SELE.NDOCTO_ELE
             AND T05.CSUCUR_T05 = SELE.CSUCUR_ELE
             AND T05.CTPDOC_T05 = SELE.CTPDOC_ELE
             AND T05.NITEM_T05 = SELE.NITEM_ELE
       JOIN CONCORDE.SNAUGRCP RCP 
       ON RCP.NSINIE_RCP = SELE.NSINIE_ELE
       JOIN CONCORDE.ABACSMAE MAE 
       ON MAE.NRUT_MAE = SIN.NRUTPR_SIN
       JOIN CONCORDE.SNACSCDT CDT ON CDT.NSINIE_CDT = SELE.NSINIE_ELE
       JOIN CONCORDE.SNACSACC SACC ON SACC.NSINIE_ACC = SELE.NSINIE_ELE
 WHERE     SELE.FINSND_ELE = 1
       AND SELE.FINSNM_ELE = 3
       AND SELE.FINSNA_ELE = 2019
       AND SELE.NSINIE_ELE = 6595333;
